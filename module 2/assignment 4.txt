B) Build a Flipkart-style discount calculator: given product price, discount percentage, and a boolean isMember, use arithmetic and logical operators to calculate the final price (apply an extra 5% off if isMember is true).

#include <stdio.h>
#include <conio.h>

void main()
{
    float price; 
    float discount = 50; 
    float finalPrice;
    int Member;

    printf("\nEnter Product Price: ");
    scanf("%f", &price);

    printf("\nthe discount in product is %f" ,discount);
    

    printf("\nAre you a Member? (1 = Yes, 0 = No): ");
    scanf("%d", &Member);

    // Apply normal discount
    finalPrice = price - (price * discount / 100);

    // Apply extra 5% discount if member
    if (Member && finalPrice > 0)
    {
        finalPrice = finalPrice - (finalPrice * 5 / 100);
    }

    printf("\nOriginal Price : %f", price);
    printf("\nDiscount       : %f", discount);

    if (Member)
    {
        printf("\nMember Discount: 5%% Extra Applied");
    }
    else
    {
        printf("\nMember Discount: Not Applied");
    }

    printf("\nFinal Price    : %f", finalPrice);

    getch();
}
